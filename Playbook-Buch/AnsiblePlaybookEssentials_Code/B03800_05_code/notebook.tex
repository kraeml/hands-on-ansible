    \section{Ablaufsteuerung}\label{ablaufsteuerung}

Flussablauf Ansible

\begin{itemize}
\item
  Bedingungen
\item
  \texttt{when}
\item
  Iterativ
\item
  \texttt{with\_*}
\item
  Verwendung von Bedingungen mit den \texttt{with\_*} Statement
\item
  Verwendung von Variablen und \texttt{facts} um Subroutinen zu
  überspringen
\item
  Selektiver Anwendung von Rollen
\item
  Bedingungen in Jinja2-Vorlagen
\end{itemize}

    \subsection{Bedingungen}\label{bedingungen}

When-Satement:

\begin{itemize}
\tightlist
\item
  Ob eine Variable definiert ist
\item
  Ob ein vorheriges Kommando erfolgreich abgelaufen ist
\item
  Ob ein Task abgelaufen ist
\item
  Ob die Zielplatform entsprechend der unterstützten Plattformen
  entspricht
\item
  Ob eine bestimmte Datei vorhanden ist
\end{itemize}
